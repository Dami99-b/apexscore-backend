const API = "https://apexscore.onrender.com"; let data = []; async function load() { try { const res = await fetch(`${API}/api/applicants?limit=50`); const json = await res.json(); if (json.applicants && json.applicants.length > 0) { data = json.applicants; render(); updateStats(); } } catch(e) { console.error('Failed to load from API:', e); document.getElementById('table').innerHTML = ` Failed to load data. Please check if API is running. `; } } function updateStats() { const total = data.length; const highRisk = data.filter(a => a.risk_level === "High").length; const avgScore = total > 0 ? Math.round(data.reduce((sum, a) => sum + a.apex_score, 0) / total) : 0; document.getElementById('totalApplicants').textContent = total; document.getElementById('highRiskCount').textContent = highRisk; document.getElementById('avgScore').textContent = avgScore; document.getElementById('tableCount').textContent = total; } function getRiskColor(level) { if (level === "Low") return "bg-green-500/20 text-green-400 border-green-500/30"; if (level === "Medium") return "bg-yellow-500/20 text-yellow-400 border-yellow-500/30"; return "bg-red-500/20 text-red-400 border-red-500/30"; } function getScoreColor(score) { if (score >= 75) return "text-green-400"; if (score >= 50) return "text-yellow-400"; return "text-red-400"; } function render() { const tbody = document.getElementById("table"); tbody.innerHTML = ""; if (data.length === 0) { tbody.innerHTML = ` No applicants found. Loading... `; return; } data.forEach(a => { const tr = document.createElement("tr"); tr.className = "border-b border-white/5 hover:bg-white/5 cursor-pointer transition-all"; tr.onclick = () => showDetail(a); tr.innerHTML = `
${a.email}
${a.name.full} ${a.location.city} ${a.apex_score} ${a.risk_level} `; tbody.appendChild(tr); }); } // Search functionality async function searchApplicant() { const email = document.getElementById('searchInput').value.trim(); if (!email) { alert('Please enter an email address'); return; } try { const res = await fetch(`${API}/api/search?email=${encodeURIComponent(email)}`); const applicant = await res.json(); if (applicant && applicant.email) { showDetail(applicant); } else { alert('Applicant not found'); } } catch(e) { console.error('Search failed:', e); alert('Search failed. Please try again.'); } } function showDetail(applicant) { const modal = document.getElementById('detailModal'); const content = document.getElementById('modalContent'); document.getElementById('modalTitle').textContent = applicant.name.full; content.innerHTML = `
${applicant.apex_score}
ApexScore Risk Rating
${applicant.risk_level} Risk ${applicant.action_recommendation.recommendation}

Personal Information

Email: ${applicant.email}
Phone: ${applicant.phone}
Occupation: ${applicant.occupation}
SIM Status: ${applicant.sim_registration}

Location Details

City: ${applicant.location.city}
Country: ${applicant.location.country}
Address: ${applicant.location.address}
Coordinates: ${applicant.location.coordinates.lat}, ${applicant.location.coordinates.lng}

Network Information

ISP: ${applicant.network.isp}
IP Address: ${applicant.network.ip_address}

Device Fingerprint

Type: ${applicant.device_fingerprint.device_type}
Model: ${applicant.device_fingerprint.model}
OS: ${applicant.device_fingerprint.os_version}
Rooted: ${applicant.device_fingerprint.is_rooted ? 'Yes' : 'No'}
VPN: ${applicant.device_fingerprint.vpn_detected ? 'Yes' : 'No'}

Behavioral Stability Indicators (BSI)

${applicant.bsi.location_consistency}
Location
${applicant.bsi.device_stability}
Device
${applicant.bsi.sim_changes}
SIM

Financial Overview

Outstanding Debt: $${applicant.tfd.outstanding_debt.toLocaleString()}
Total Loans: ${applicant.tfd.loan_history.length}
${applicant.tfd.loan_history.length > 0 ? `

Loan History

${applicant.tfd.loan_history.map(loan => `
${loan.institution}
${loan.loan_id}
${loan.status}
Amount: $${loan.amount.toLocaleString()}
Purpose: ${loan.purpose}
Disbursed: ${loan.disbursement_date}
Due: ${loan.due_date}
${loan.days_overdue ? `
Days Overdue: ${loan.days_overdue}
` : ''}
`).join('')}
` : ''}

Bank Accounts

${applicant.bank_accounts.map(acc => `
${acc.bank_name}
${acc.account_number}
${acc.status}
`).join('')}
`; modal.classList.remove('hidden'); } // Event Listeners document.getElementById('searchBtn').addEventListener('click', searchApplicant); document.getElementById('searchInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') searchApplicant(); }); document.getElementById('closeModal').addEventListener('click', () => { document.getElementById('detailModal').classList.add('hidden'); }); document.getElementById('detailModal').addEventListener('click', (e) => { if (e.target.id === 'detailModal') { document.getElementById('detailModal').classList.add('hidden'); } }); document.getElementById('statsBtn').addEventListener('click', () => { alert(`Total: ${data.length}\nHigh Risk: ${data.filter(a => a.risk_level === "High").length}\nAvg Score: ${Math.round(data.reduce((sum, a) => sum + a.apex_score, 0) / data.length)}`); }); // Auto-refresh table every 3 seconds setInterval(() => { load(); }, 3000); // Initial load load();
